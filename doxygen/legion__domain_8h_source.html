262  if(a.dim > b.dim) return false;
263  for(int i = 0; (i == 0) || (i < a.dim); i++) {
264  if(a.point_data[i] < b.point_data[i]) return true;
265  if(a.point_data[i] > b.point_data[i]) return false;
266  }
267  return false;
