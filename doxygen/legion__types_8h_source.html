351 // single normal or a single reduction instance
352 COLLECTIVE_REDUCTION = 3,

354 // to a collective normal or reduction instance using a butterfly
355 // network reduction (both instances using the same nodes)
356 COLLECTIVE_BUTTERFLY_ALLREDUCE = 4,
357 // Performing an all-reduce by doing a reduction down to a single
358 // instance and then broadcasting the result out from that instance
359 // (instances don't exist on the same set of nodes)

361 // Copy from one collective normal instanace to another collective

368 // instance to any kind of other instance without doing an all-reduce
369 COLLECTIVE_HAMMER_REDUCTION = 8,

699 // Methodology for assigning priorities to meta-tasks:
700 // Minimum and low priority are for things like profiling
701 // that we don't want to interfere with normal execution.
702 // Resource priority is reserved for tasks that have been
703 // granted resources like reservations. Running priority
704 // is the highest and guarantees that we drain out any
705 // previously running tasks over starting new ones. The rest
706 // of the priorities are classified as either 'throughput'
707 // or 'latency' sensitive. Under each of these two major
708 // categories there are four sub-priorities:
709 // - work: general work to be done
710 // - deferred: work that was already scheduled but
711 // for which a continuation had to be
712 // made so we don't want to wait behind
713 // work that hasn't started yet
714 // - messsage: a message from a remote node that we
715 // should handle sooner than our own
716 // work since work on the other node is
717 // blocked waiting on our response
718 // - response: a response message from a remote node
719 // that we should handle to unblock work
720 // on our own node
721 enum LgPriority {
722 LG_MIN_PRIORITY = INT_MIN,
723 LG_LOW_PRIORITY = -1,
724 // Throughput priorities
725 LG_THROUGHPUT_WORK_PRIORITY = 0,
726 LG_THROUGHPUT_DEFERRED_PRIORITY = 1,
727 LG_THROUGHPUT_MESSAGE_PRIORITY = 2,
728 LG_THROUGHPUT_RESPONSE_PRIORITY = 3,
729 // Latency priorities
730 LG_LATENCY_WORK_PRIORITY = 4,
731 LG_LATENCY_DEFERRED_PRIORITY = 5,
732 LG_LATENCY_MESSAGE_PRIORITY = 6,
733 LG_LATENCY_RESPONSE_PRIORITY = 7,
734 // Resource priorities
735 LG_RESOURCE_PRIORITY = 8,
736 // Running priorities
737 LG_RUNNING_PRIORITY = 9,
738 };
