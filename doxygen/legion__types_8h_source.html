581  // Methodology for assigning priorities to meta-tasks:
582  // Minimum and low priority are for things like profiling
583  // that we don't want to interfere with normal execution.
584  // Resource priority is reserved for tasks that have been
585  // granted resources like reservations. Running priority
586  // is the highest and guarantees that we drain out any
587  // previously running tasks over starting new ones. The rest
588  // of the priorities are classified as either 'throughput'
589  // or 'latency' sensitive. Under each of these two major
590  // categories there are four sub-priorities:
591  // - work: general work to be done
592  // - deferred: work that was already scheduled but
593  // for which a continuation had to be
594  // made so we don't want to wait behind
595  // work that hasn't started yet
596  // - messsage: a message from a remote node that we
597  // should handle sooner than our own
598  // work since work on the other node is
599  // blocked waiting on our response
600  // - response: a response message from a remote node
601  // that we should handle to unblock work
602  // on our own node
603  enum LgPriority {
604  LG_MIN_PRIORITY = INT_MIN,
605  LG_LOW_PRIORITY = -1,
606  // Throughput priorities
607  LG_THROUGHPUT_WORK_PRIORITY = 0,
608  LG_THROUGHPUT_DEFERRED_PRIORITY = 1,
609  LG_THROUGHPUT_MESSAGE_PRIORITY = 2,
610  LG_THROUGHPUT_RESPONSE_PRIORITY = 3,
611  // Latency priorities
612  LG_LATENCY_WORK_PRIORITY = 4,
613  LG_LATENCY_DEFERRED_PRIORITY = 5,
614  LG_LATENCY_MESSAGE_PRIORITY = 6,
615  LG_LATENCY_RESPONSE_PRIORITY = 7,
